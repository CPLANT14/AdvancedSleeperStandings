  const [hoveredYear, setHoveredYear] = useState(null);

  const hoverTimeoutRef = useRef(null);

  

  // Results & Sorting & Filtering State

  const [standings, setStandings] = useState([]);

  const globalUserCache = useRef({});


  const getPowerColor = (score) => {

    if (score >= 80) return 'bg-blue-500';

    if (score >= 60) return 'bg-emerald-500';

    if (score >= 40) return 'bg-yellow-500';

    return 'bg-rose-600';

  };


  const getAllPlayColor = (pct) => {

    if (pct >= 0.70) return 'bg-blue-900/40 text-blue-400 border-blue-800/50';

    if (pct >= 0.50) return 'bg-emerald-900/40 text-emerald-400 border-emerald-800/50';

    if (pct >= 0.35) return 'bg-yellow-900/40 text-yellow-400 border-yellow-800/50';

    return 'bg-rose-900/40 text-rose-400 border-rose-800/50';

  };


  const getWoeColor = (woe) => {

    if (woe >= 1.5) return 'text-blue-400';

    if (woe > 0) return 'text-emerald-400';

    if (woe > -1.5) return 'text-yellow-500';

    return 'text-rose-600';

  };


  const getRecordColor = (w, l) => {

    if (w > l) return 'text-emerald-400 bg-emerald-950/20 px-2 py-0.5 rounded-md border border-emerald-900/30';

    if (l > w) return 'text-rose-400 bg-rose-950/10 px-2 py-0.5 rounded-md border border-rose-900/20';

    return 'text-gray-400';

  };


  const getPfRelColor = (val) => {

    const threshold = perGameMode ? 10 : 100;

    if (val > threshold) return 'bg-emerald-500/10 text-emerald-400';

    if (val > 0) return 'bg-emerald-500/5 text-emerald-500';

    if (val < -threshold) return 'bg-rose-500/10 text-rose-400';

    if (val < 0) return 'bg-rose-500/5 text-rose-500';

    return 'text-white';

  };


  const applyPreset = (p) => {

    setRangeStart(p.start);

    setRangeEnd(p.end);

    setInterval(p.end - p.start + 1);

  };


  const scrollToGlossary = () => {

    const glossary = document.getElementById('glossary-section');

    if (glossary) {

      glossary.scrollIntoView({ behavior: 'smooth' });

    }

  };


  const getGeneratedSegments = () => {

    const segments = [];

    const step = Math.max(1, interval);

    for (let i = rangeStart; i <= rangeEnd; i += step) {

      const end = Math.min(i + step - 1, rangeEnd);

      segments.push({ start: i, end: end });

    }

    return segments;

  };


  const copyTableToClipboard = () => {

    if (!standings.length) return;

    const sortedData = getSortedStandings();

    const headers = [

      "League", "Season", "Range", "Rank", "Team", "Manager", "Wins", "Losses", 

      perGameMode ? "PF/G" : "PF", 

      perGameMode ? "PA/G" : "PA", 

      perGameMode ? "PF +/- (G)" : "PF +/-",

      "Power Score", "All-Play %", "xW", "WOE", "Playoff Finish"

    ];

    let tsvContent = headers.join("\t") + "\n";

    sortedData.forEach((team) => {

      const games = (team.weeksIncluded || 1);

      const row = [

        team.leagueName, team.season, team.segmentLabel, team.rank, team.teamName, team.userName,

        team.wins, team.losses, 

        (perGameMode ? team.pf / games : team.pf).toFixed(2),

        (perGameMode ? team.pa / games : team.pa).toFixed(2),

        (perGameMode ? team.pfRel / games : team.pfRel).toFixed(2),

        team.powerScore.toFixed(0),

        (team.allPlayPct * 100).toFixed(1) + "%", team.xW.toFixed(2), team.woe.toFixed(2),

        team.playoffFinish || "-"

      ];

      tsvContent += row.join("\t") + "\n";

    });

    const textArea = document.createElement("textarea");

    textArea.value = tsvContent;

    document.body.appendChild(textArea);

    textArea.select();

    try {

      document.execCommand('copy');

      setCopied(true);

      setTimeout(() => setCopied(false), 2000);

    } catch (err) {}

    document.body.removeChild(textArea);

  };


  const handleUserSearch = async (e) => {

    if (e) e.preventDefault();

    if (!username) return;

    setLoading(true);

    setError(null);

    try {

      const userRes = await fetch(`https://api.sleeper.app/v1/user/${username.trim()}`);

      if (!userRes.ok) throw new Error("User not found.");

      const userData = await userRes.json();

      setSelectedUser(userData);

      setStandings([]);

      setLeaguesByYear({});

      setSelectedLeagueIds(new Set());

      setIsConfiguring(true);

      await fetchLeaguesForYear('2025', userData.user_id, true);

    } catch (err) {

      setError(err.message);

    } finally {

      setLoading(false);

    }

  };


  const fetchLeaguesForYear = async (year, userId = selectedUser?.user_id, autoSelect = false) => {

    if (leaguesByYear[year]) return;

    setLoading(true);

    try {

      const res = await fetch(`https://api.sleeper.app/v1/user/${userId}/leagues/nfl/${year}`);

      const data = await res.json();

      setLeaguesByYear(prev => ({ ...prev, [year]: data }));

      

      if (autoSelect) {

        setSelectedLeagueIds(prev => {

          const next = new Set(prev);

          data.forEach(l => next.add(l.league_id));

          return next;

        });

      }

    } catch (err) {

      console.error(err);

    } finally {

      setLoading(false);

    }

  };


  const toggleLeagueSelection = (id) => {

    const next = new Set(selectedLeagueIds);

    if (next.has(id)) next.delete(id);

    else next.add(id);

    setSelectedLeagueIds(next);

  };


  const selectYearAll = (year) => {

    const yearLeagues = leaguesByYear[year] || [];

    const next = new Set(selectedLeagueIds);

    yearLeagues.forEach(l => next.add(l.league_id));

    setSelectedLeagueIds(next);

  };


  const deselectYearAll = (year) => {

    const yearLeagues = leaguesByYear[year] || [];

    const next = new Set(selectedLeagueIds);

    yearLeagues.forEach(l => next.delete(l.league_id));

    setSelectedLeagueIds(next);

  };


  const handleYearMouseEnter = (year) => {

    if (hoverTimeoutRef.current) clearTimeout(hoverTimeoutRef.current);

    setHoveredYear(year);

    if (!leaguesByYear[year]) {

        fetchLeaguesForYear(year, selectedUser?.user_id, false);

    }

  };


  const handleYearMouseLeave = () => {

    hoverTimeoutRef.current = setTimeout(() => {

      setHoveredYear(null);

    }, 300);

  };


  const resolveManagerData = async (userId, currentLeagueUsersMap) => {

    if (!userId || userId === "0") return { displayName: "Inactive", teamName: "Empty Roster" };

    if (currentLeagueUsersMap[userId]) return currentLeagueUsersMap[userId];

    if (globalUserCache.current[userId]) return globalUserCache.current[userId];

    

    try {

      const res = await fetch(`https://api.sleeper.app/v1/user/${userId}`);

      if (res.ok) {

        const data = await res.json();

        const resolved = { 

          displayName: data.display_name || "Former Manager", 

          teamName: data.metadata?.team_name || `${data.display_name || 'Former'}'s Team`

        };

        globalUserCache.current[userId] = resolved;

        return resolved;

      }

    } catch (e) { console.error("Failed to fetch global user", userId); }

    return { displayName: "Unknown Manager", teamName: "Unknown Team" };

  };


  const fetchStats = async () => {

    if (selectedLeagueIds.size === 0) return;

    setLoading(true);

    setError(null);

    

    try {

      const segments = getGeneratedSegments();

      let allResults = [];

      

      const flatLeagues = Object.entries(leaguesByYear).flatMap(([year, leagues]) => 

        leagues.map(l => ({ ...l, year }))

      ).filter(l => selectedLeagueIds.has(l.league_id));


      for (const league of flatLeagues) {

        const [usersRes, rostersRes, winnersRes] = await Promise.all([

          fetch(`https://api.sleeper.app/v1/league/${league.league_id}/users`).then(r => r.json()),

          fetch(`https://api.sleeper.app/v1/league/${league.league_id}/rosters`).then(r => r.json()),

          fetch(`https://api.sleeper.app/v1/league/${league.league_id}/winners_bracket`).then(r => r.json()).catch(() => [])

        ]);


        const leagueUserMap = usersRes.reduce((acc, u) => {

          acc[u.user_id] = { 

            displayName: u.display_name, 

            teamName: u.metadata?.team_name || u.metadata?.nickname || `${u.display_name}'s Team` 

          };

          return acc;

        }, {});


        const rosterToOwnerData = {};

        const managerResolutionPromises = rostersRes.map(async (roster) => {

          rosterToOwnerData[roster.roster_id] = await resolveManagerData(roster.owner_id, leagueUserMap);

        });

        await Promise.all(managerResolutionPromises);


        const finishMap = {};

        if (Array.isArray(winnersRes)) {

            winnersRes.forEach(match => {

              if (match.p) {

                if (match.w) finishMap[match.w] = match.p;

                if (match.l) finishMap[match.l] = match.p + 1; 

              }

            });

        }


        const allRequiredWeeks = Array.from({ length: rangeEnd - rangeStart + 1 }, (_, i) => rangeStart + i);

        const rawWeeksData = await Promise.all(

          allRequiredWeeks.map(w => 

            fetch(`https://api.sleeper.app/v1/league/${league.league_id}/matchups/${w}`)

              .then(res => res.ok ? res.json() : [])

              .catch(() => [])

          )

        );

        

        const weekDataMap = allRequiredWeeks.reduce((acc, w, i) => {

          acc[w] = rawWeeksData[i];

          return acc;

        }, {});


        const isMedianLeague = league.settings?.league_average_match === 1;


        segments.forEach((segment) => {

          const leagueStats = rostersRes.reduce((acc, r) => {

            acc[r.roster_id] = { wins: 0, losses: 0, pf: 0, pa: 0, allPlayWins: 0, allPlayTotal: 0, weeksPlayed: 0 };

            return acc;

          }, {});


          for (let w = segment.start; w <= segment.end; w++) {

            let weekMatchups = weekDataMap[w];

            if (!weekMatchups?.length) continue;


            const matchupGroupMap = {};

            weekMatchups.forEach(m => {

              if (leagueStats[m.roster_id]) {

                leagueStats[m.roster_id].pf += (m.points || 0);

                leagueStats[m.roster_id].weeksPlayed++;

                if (!matchupGroupMap[m.matchup_id]) matchupGroupMap[m.matchup_id] = [];

                matchupGroupMap[m.matchup_id].push(m);

              }

            });


            Object.values(matchupGroupMap).forEach(pair => {

              if (pair.length === 2) {

                const [t1, t2] = pair;

                if (leagueStats[t1.roster_id] && leagueStats[t2.roster_id]) {

                    leagueStats[t1.roster_id].pa += (t2.points || 0);

                    leagueStats[t2.roster_id].pa += (t1.points || 0);

                    if (t1.points > t2.points) { leagueStats[t1.roster_id].wins++; leagueStats[t2.roster_id].losses++; }

                    else if (t2.points > t1.points) { leagueStats[t2.roster_id].wins++; leagueStats[t1.roster_id].losses++; }

                }

              }

            });


            if (includeMedian && isMedianLeague) {

              const sortedScores = [...weekMatchups].sort((a, b) => (b.points || 0) - (a.points || 0));

              const halfSize = Math.floor(sortedScores.length / 2);

              sortedScores.forEach((m, idx) => {

                if (leagueStats[m.roster_id]) {

                    if (idx < halfSize) leagueStats[m.roster_id].wins++;

                    else leagueStats[m.roster_id].losses++;

                }

              });

            }


            weekMatchups.forEach(t1 => {

              if (!leagueStats[t1.roster_id]) return;

              weekMatchups.forEach(t2 => {

                if (t1.roster_id !== t2.roster_id) {

                  leagueStats[t1.roster_id].allPlayTotal++;

                  if (t1.points > t2.points) leagueStats[t1.roster_id].allPlayWins++;

                }

              });

              if (includeMedian && isMedianLeague) {

                  const sortedScores = [...weekMatchups].sort((a, b) => (b.points || 0) - (a.points || 0));

                  const halfSize = Math.floor(sortedScores.length / 2);

                  const isTopHalf = sortedScores.findIndex(m => m.roster_id === t1.roster_id) < halfSize;

                  leagueStats[t1.roster_id].allPlayTotal++;

                  if (isTopHalf) leagueStats[t1.roster_id].allPlayWins++;

              }

            });

          }


          const pfArray = rostersRes.map(r => leagueStats[r.roster_id].pf);

          const paArray = rostersRes.map(r => leagueStats[r.roster_id].pa);

          const leagueMeanPf = pfArray.reduce((a, b) => a + b, 0) / (pfArray.length || 1);

          const leagueMeanPa = paArray.reduce((a, b) => a + b, 0) / (paArray.length || 1);

          const leagueStdDevPf = Math.sqrt(pfArray.map(x => Math.pow(x - leagueMeanPf, 2)).reduce((a, b) => a + b, 0) / (pfArray.length || 1)) || 1;


          const segmentResults = rostersRes.map(roster => {

            const manager = rosterToOwnerData[roster.roster_id] || { displayName: "Unknown", teamName: "Unknown Team" };

            const stats = leagueStats[roster.roster_id];

            const gamesPlayed = stats.wins + stats.losses;

            const allPlayPct = stats.allPlayTotal > 0 ? stats.allPlayWins / stats.allPlayTotal : 0;

            const xW = allPlayPct * gamesPlayed;


            const zScore = (stats.pf - leagueMeanPf) / leagueStdDevPf;

            const powerScore = Math.min(99, Math.max(0, 50 + (zScore * 20)));


            return {

              leagueId: league.league_id,

              leagueName: league.name,

              season: league.year,

              segmentLabel: segment.start === segment.end ? `Week ${segment.start}` : `${segment.start}-${segment.end}`,

              weeksIncluded: stats.weeksPlayed,

              userName: manager.displayName,

              teamName: manager.teamName,

              wins: stats.wins,

              losses: stats.losses,

              pf: stats.pf,

              pa: stats.pa,

              leagueMeanPf,

              leagueMeanPa,

              pfRel: stats.pf - leagueMeanPf,

              powerScore: powerScore,

              allPlayPct: allPlayPct,

              xW: xW,

              woe: stats.wins - xW,

              playoffFinish: finishMap[roster.roster_id] || null,

              isMedianActive: includeMedian && isMedianLeague

            };

          });


          segmentResults.sort((a, b) => b.wins - a.wins || b.pf - a.pf);

          allResults = [...allResults, ...segmentResults.map((t, i) => ({ ...t, rank: i + 1 }))];

        });

      }


      setStandings(allResults);

      setIsConfiguring(false);

    } catch (e) {

      console.error(e);

      setError("Failed to fetch league data. Please try again.");

    } finally {

      setLoading(false);

    }

  };


  const requestSort = (key) => {

    let direction = 'desc';

    if (sortConfig.key === key && sortConfig.direction === 'desc') {

      direction = 'asc';

    }

    setSortConfig({ key, direction });

  };


  const getFilteredStandings = () => {

    if (!managerFilter) return standings;

    const lower = managerFilter.toLowerCase();

    return standings.filter(s => 

      s.userName.toLowerCase().includes(lower) || 

      s.teamName.toLowerCase().includes(lower) ||

      s.leagueName.toLowerCase().includes(lower)

    );

  };


  const getSortedStandings = () => {

    const filtered = getFilteredStandings();

    if (!sortConfig.key) return filtered;

    return [...filtered].sort((a, b) => {

      let aVal = a[sortConfig.key];

      let bVal = b[sortConfig.key];

      if (aVal < bVal) return sortConfig.direction === 'asc' ? -1 : 1;

      if (aVal > bVal) return sortConfig.direction === 'asc' ? 1 : -1;

      return 0;

    });

  };


  const SortIndicator = ({ columnKey }) => {

    if (sortConfig.key !== columnKey) return null;

    return sortConfig.direction === 'asc' ? 

      <ArrowUpNarrowWide className="w-2.5 h-2.5 text-white ml-1" /> : 

      <ArrowDownWideNarrow className="w-2.5 h-2.5 text-white ml-1" />;

  };


  const HeaderWithGlossaryLink = ({ label, sortKey, className = "", align = "center" }) => {

    const alignClass = align === "left" ? "justify-start" : align === "right" ? "justify-end" : "justify-center";

    const isSorted = sortConfig.key === sortKey;

    

    return (

      <th className={`px-4 py-4 select-none ${className}`}>

        <div className={`flex items-center gap-1 ${alignClass}`}>

          <button 

            onClick={() => sortKey && requestSort(sortKey)}

            className={`flex items-center gap-1 group/btn hover:opacity-80 transition-opacity ${sortKey ? 'cursor-pointer' : 'cursor-default'}`}

          >

            <span className={`text-[9px] font-black uppercase tracking-widest whitespace-nowrap transition-colors ${isSorted ? 'text-white' : 'text-gray-400'}`}>

              {label}

            </span>

            <SortIndicator columnKey={sortKey} />

          </button>

          <button 

            onClick={scrollToGlossary}

            className="p-1 hover:bg-emerald-900/50 rounded-full transition-colors group"

            title={`What is ${label}?`}

          >
